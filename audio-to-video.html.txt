function loadSrc(src){ return new Promise((ok,fail)=>{ if(document.querySelector(`script[src="${src}"]`)){ok();return;} const s=document.createElement('script'); s.src=src; s.crossOrigin='anonymous'; s.onload=ok; s.onerror=()=>fail(new Error('Cannot load: '+src)); document.head.appendChild(s); }); }
function showEng(msg,err=false){ const e=$('engStatus'); e.innerHTML=msg; e.className='engine-status show'+(err?' err':''); }

async function initEngine(){
  if(ffReady||ffLoading) return; ffLoading=true;
  showEng('<span class="spinner"></span> Loading video engine (~30 MB)…');
  try{
    let ok=false;
    for(const [a,b] of [
      ['https://unpkg.com/@ffmpeg/ffmpeg@0.12.10/dist/umd/ffmpeg.js','https://unpkg.com/@ffmpeg/util@0.12.1/dist/umd/index.js'],
      ['https://cdn.jsdelivr.net/npm/@ffmpeg/ffmpeg@0.12.10/dist/umd/ffmpeg.js','https://cdn.jsdelivr.net/npm/@ffmpeg/util@0.12.1/dist/umd/index.js']
    ]){ try{ await loadSrc(a); await loadSrc(b); ok=true; break; }catch(e){ console.warn(e.message); } }
    if(!ok) throw new Error('CDN load failed');
    ffInst=new FFmpegWASM.FFmpeg();
    ffInst.on('log',({message})=>{ const l=$('progLog'); l.textContent+=message+'\n'; l.scrollTop=l.scrollHeight; });
    ffInst.on('progress',({progress})=>{ const p=Math.min(99,Math.round(progress*100)); $('progFill').style.width=p+'%'; $('progPct').textContent=p+'%'; });
    let core=false;
    for(const base of ['https://unpkg.com/@ffmpeg/core@0.12.6/dist/umd','https://cdn.jsdelivr.net/npm/@ffmpeg/core@0.12.6/dist/umd']){
      try{ await ffInst.load({ coreURL:await FFmpegUtil.toBlobURL(base+'/ffmpeg-core.js','text/javascript'), wasmURL:await FFmpegUtil.toBlobURL(base+'/ffmpeg-core.wasm','application/wasm') }); core=true; break; }catch(e){ console.warn(e.message); }
    }
    if(!core) throw new Error('Core load failed');
    ffReady=true; hideEng(); if(selFile) $('cvtBtn').disabled=false;
  }catch(err){
    showEng('⚠️ Video engine failed to load.<br><span style="font-size:10px;opacity:0.8;">Save this file to your device and open it locally (Files app on iPad, file manager on Android).</span>',true);
    console.error(err);
  }
  ffLoading=false;
}

function drawCover(canvas,text,bg,fg,fss){ const ctx=canvas.getContext('2d'); const W=canvas.width,H=canvas.height; ctx.fillStyle=bg; ctx.fillRect(0,0,W,H); const fs=fss==='auto'?Math.max(20,Math.min(90,Math.floor(W/18))):parseInt(fss); ctx.fillStyle=fg; ctx.textAlign='center'; ctx.textBaseline='middle'; ctx.font=`${fs}px "Segoe UI","Helvetica Neue",Arial,sans-serif`; const lines=wrapText(ctx,text,W*0.82,fs); const lh=fs*1.45; const sy=H/2-(lines.length*lh)/2+lh/2; lines.forEach((l,i)=>ctx.fillText(l,W/2,sy+i*lh)); }
function wrapText(ctx,text,maxW,fs){ ctx.font=`${fs}px "Segoe UI","Helvetica Neue",Arial,sans-serif`; const words=text.split(' '),lines=[]; let cur=''; for(const w of words){ const t=cur?cur+' '+w:w; if(ctx.measureText(t).width<=maxW){cur=t;}else{ if(cur)lines.push(cur); if(ctx.measureText(w).width>maxW){let p=''; for(const ch of w){if(ctx.measureText(p+ch).width>maxW){lines.push(p);p=ch;}else p+=ch;} cur=p;}else cur=w; } } if(cur)lines.push(cur); return lines; }

    const data=await ffInst.readFile('output.mp4');
    const blob=new Blob([data.buffer],{type:'video/mp4'});
